require "godot"

type UserName = String
type GreetingText = String
type MaybeGreeting = String | nil

module NameFormatter
  def format_name(name: UserName): UserName
    return name.capitalize
  end
end

class Greeter
  include NameFormatter

  @name: UserName
  @punctuation: String = "!"

  attr_reader :name
  attr_accessor :punctuation

  def initialize(name: UserName, punctuation: String = "!"): void
    @name = format_name(name)
    @punctuation = punctuation
  end

  def salute: GreetingText
    return "Hello " + @name + @punctuation
  end

  def maybe_salute(enabled: Boolean): MaybeGreeting
    if !enabled
      return nil
    end
    return salute
  end
end

class Main < Node
  @label: Label

  def ready: void
    greeter: Greeter = Greeter.new("world")
    @label = Label.new
    @label.text = greeter.salute
    add_child(@label)
  end
end
